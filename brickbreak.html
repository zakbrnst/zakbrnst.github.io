from tkinter import *
import time
import winsound
from random import randint

tk = Tk()

#variables
windHeight = 500
windWidth = 600
addX = 25
addY = 25
x = 100
y = 100
xv = 1
yv = 1
paddleX = 150
paddleY = windHeight - 10
paddleWidth = 50
score = 0
brickX = randint(100, windWidth - 100)
brickY = randint(100, windHeight - 100)
brickWidth = randint(50, 150)
brickHeight = randint(50,150)

def keyInput(i):
global paddleX
if i.keysym == "Left":
paddleX = paddleX - 5
if i.keysym == "Right":
paddleX = paddleX + 5

myLabel = Label(tk, text = "Hey Gamers")
myLabel.pack()
wind = Canvas (tk, bg = "#5394fc", height = windHeight, width = windWidth)
wind.pack()


wind.bind("", keyInput)
wind.focus_set()
wind.pack()



while True:
x = x + xv
y = y+yv
if x < 0 or x > windWidth - addX:
xv = xv*-1
if y < 0 or y > windHeight - addY:
score = score - 1
yv = yv*-1
if x > paddleX and x < paddleX + paddleWidth and y+addY > paddleY and y+addY < paddleY + 10:
yv = yv*-1
if x > brickX and x < brickX + brickWidth and y+addY > brickY and y+addY < brickY + brickHeight:
yv = yv*-1
brickX = randint(100, windWidth - 100)
brickY = randint(100, windHeight - 100)
brickWidth = randint(50, 150)
brickHeight = randint(50, 150)


wind.delete("all")
ball = wind.create_oval(x, y, x+addX, y+addY, fill = "white")
wind.create_rectangle(paddleX, paddleY, paddleX+paddleWidth, paddleY - 10, fill = "black")
wind.create_rectangle(brickX, brickY, brickX + brickWidth, brickY + brickHeight, fill = "red")
tk.update()
time.sleep(0.01)


tk.mainloop()
